/*
 * helloworld.c: simple test application
 *
 * This application configures UART 16550 to baud rate 9600.
 * PS7 UART (Zynq) is not initialized by this application, since
 * bootrom/bsp configures it to baud rate 115200
 *
 * ------------------------------------------------
 * | UART TYPE   BAUD RATE                        |
 * ------------------------------------------------
 *   uartns550   9600
 *   uartlite    Configurable only in HW design
 *   ps7_uart    115200 (configured by bootrom/bsp)
 */

#include <stdio.h>
#include "platform.h"
#include "xil_printf.h"
#include "xuartlite.h"
#include "xparameters.h"

XUartLite_Config *uart0_conf;
XUartLite_Config *uart1_conf;
XUartLite uart0_ins;
XUartLite uart1_ins;


void init(){
	uart0_conf = XUartLite_LookupConfig ( XPAR_AXI_UARTLITE_0_DEVICE_ID );
	uart1_conf = XUartLite_LookupConfig ( XPAR_AXI_UARTLITE_1_DEVICE_ID );
	int status0 = XUartLite_CfgInitialize( &uart0_ins, uart0_conf,uart0_conf->RegBaseAddr);
	int status1 = XUartLite_CfgInitialize( &uart1_ins, uart1_conf,uart1_conf->RegBaseAddr);
	if ((status0 && status1)==XST_SUCCESS){
		xil_printf("Success initialize\r\n");
	}
	else{
		xil_printf("Error initialize\r\n");
	}
}

int main()
{
    init_platform();
    init();
    char data0[10]="Salam1234567";
    char received;
    XUartLite_Send(&uart0_ins, data0,10);
    //while( XUartLite_IsSending(&uart0_ins));

    while(1){
    unsigned int count = XUartLite_Recv(&uart1_ins, &received,1);
    xil_printf("%c",received);
    sleep(1);
    }
    cleanup_platform();
    return 0;
}
